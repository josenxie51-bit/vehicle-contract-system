# 单辆车查询函数示例
def fetch_single_vehicle_data(plate_number, plate_type="", contract_number=""):
    """
    查询单辆车的合同备案数据
    """
    url = "https://cq.122.gov.cn/vio/rent/contract/list"
    
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8',
        'Referer': 'https://cq.122.gov.cn/views/memzl/index2.html'
    }
    
    data = {
        'hpzl': plate_type,
        'hphm': plate_number,
        'htbh': contract_number,
        'page': '1',
        'size': '10'
    }
    
    try:
        response = requests.post(url, headers=headers, data=data, timeout=30, verify=False)
        result = response.json()
        
        if 'data' in result and 'content' in result['data']:
            return result['data']['content']
        else:
            return []
            
    except Exception as e:
        print(f"查询失败: {e}")
        return []
